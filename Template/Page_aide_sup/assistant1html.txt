  <script>
   document.querySelector('.texte').addEventListener('keydown', function(e) {
    if (e.key === 'Backspace' || e.key === 'Delete') {
        // Définir un petit délai pour laisser le navigateur mettre à jour le DOM si la touche est pressée
        setTimeout(() => {
            let pTags = this.querySelectorAll('p');
            // Vérifier s'il reste des balises <p>
            if (pTags.length === 0) {
                // Empêcher la suppression si c'est la dernière balise <p>
                // Et restaurer une balise <p> vide
                document.execCommand('formatBlock', false, 'p');
                e.preventDefault();
            }
        }, 0);
    }
});
  </script>

    $(document).ready(function() {
      const btnReset = $('#btn-reset');
      const texte = $('.texte');

      const contenuInitial = `
      <p>3</p>
                <p>5</p>
                <p>1</p>     `
      ;

      btnReset.on('click', function() {
        texte.html(contenuInitial);
      });

      var editor = ace.edit("editor");
      editor.setTheme("ace/theme/monokai");
      editor.session.setMode("ace/mode/python");

      $('#btn-executer').click(function() {
        var code = editor.getValue();
        var contenu = texte.html();

        $.ajax({
          url: 'Execution.php',
          type: 'POST',
          data: { code: code, contenu: contenu },
          success: function(response) {
            var elements = response.split("\n");
            $('#resultat').text("")
            elements.forEach(function(element) {
            $('#resultat').append(element + "<br/>");
         });
          },
          error: function(xhr, status, error) {
            console.error('Erreur AJAX : ', status, error);
          }
        });
      });

      $('#btn-verifier').click(function() {
